### Example Implementation: Integrating ChatGPT API

Here’s a simple example of how you might integrate the ChatGPT API into your Node.js backend to create a coding assistant:

#### Step 1: Set Up OpenAI API

1. **Install the OpenAI SDK**:

```bash
npm install openai
```

2. **Create a Service to Handle API Calls**:

```javascript
// openaiService.js
const { Configuration, OpenAIApi } = require('openai');

const configuration = new Configuration({
    apiKey: process.env.OPENAI_API_KEY, // Set your OpenAI API key in environment variables
});
const openai = new OpenAIApi(configuration);

async function generateCode(prompt) {
    try {
        const response = await openai.createCompletion({
            model: 'text-davinci-003',
            prompt: prompt,
            max_tokens: 150,
        });
        return response.data.choices[0].text.trim();
    } catch (error) {
        console.error('Error generating code:', error);
        throw error;
    }
}

module.exports = { generateCode };
```

#### Step 2: Create an API Endpoint in Your Express App

```javascript
// server.js (or your main server file)
const express = require('express');
const { generateCode } = require('./openaiService');

const app = express();
app.use(express.json());

app.post('/api/generate-code', async (req, res) => {
    const { prompt } = req.body;
    try {
        const code = await generateCode(prompt);
        res.json({ code });
    } catch (error) {
        res.status(500).json({ error: 'Failed to generate code' });
    }
});

const PORT = process.env.PORT || 5000;
app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
});
```

#### Step 3: Build a Simple React Component to Interact with the API

```javascript
import React, { useState } from 'react';
import axios from 'axios';

const CodeGenerator = () => {
    const [prompt, setPrompt] = useState('');
    const [generatedCode, setGeneratedCode] = useState('');

    const handleGenerateCode = async () => {
        const response = await axios.post('/api/generate-code', { prompt });
        setGeneratedCode(response.data.code);
    };

    return (
        <div>
            <h1>AI Code Generator</h1>
            <textarea 
                rows="4" 
                cols="50" 
                value={prompt} 
                onChange={(e) => setPrompt(e.target.value)} 
                placeholder="Describe what you want the code to do..." 
            />
            <button onClick={handleGenerateCode}>Generate Code</button>
            <h2>Generated Code:</h2>
            <pre>{generatedCode}</pre>
        </div>
    );
};

export default CodeGenerator;
```
